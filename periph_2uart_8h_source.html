359 {
360  BSPACM_CORE_SAVED_INTERRUPT_STATE(istate);

362 
363  if (usp) {

368  rv |= eBSPACMperiphUARTfifoState_DRTX;
369  }
370  } while (0);
371  BSPACM_CORE_REENABLE_INTERRUPT(istate);
372  }
373  return rv;
374 }
