202 iBSPACMdeviceTM4CgpioPortTagFromShift (unsigned int shift)
203 {
204  int tag = 'A' + shift;
205  if (('O' - 2) < tag) {
206  tag += 2;
207  } else if (('I' - 1) < tag) {
208  tag += 1;
209  }
210  return tag;
211 }
