246 static BSPACM_CORE_INLINE
247 int
248 fifo_push_head (sFIFO * fp, uint8_t v)
249 {
250  uint16_t h = fp->head;
251  uint16_t t = fp->tail;
252  int rv = (h != t);
253 
254  fp->cell[h] = v;
255  h = FIFO_ADJUST_OFFSET(fp, 1U + h);
256  fp->head = h;
257  if (h == t) {
258  (void)fifo_pop_tail(fp, 1);
259  rv = -1;
260  }
261  return rv;
262 }
